
-- consulta datos de lote 
SELECT DISTINCT TOP (100) PERCENT
    RTRIM(dbo.lotes.nombre) AS Lote,
    RTRIM(dbo.empresas.descripcion) AS Empresa,
	dbo.fnProveedor(dbo.lotes.cod_lote) AS Proveedor,
	RTRIM(dbo.especies.descripcion) AS Especie,
    MMPPOrigen.nombre AS CorteInicial,
    UPPER(dbo.subproceso.nombre) AS Proceso,
    dbo.lotes.cod_lote AS IdLote
FROM dbo.lotes
LEFT OUTER JOIN dbo.especies ON dbo.lotes.cod_especie = dbo.especies.cod_especie
LEFT OUTER JOIN dbo.corte AS MMPPOrigen ON dbo.lotes.cod_corte = MMPPOrigen.cod_corte
LEFT OUTER JOIN dbo.empresas ON dbo.lotes.cod_empresa = dbo.empresas.cod_empresa
LEFT OUTER JOIN dbo.subproceso ON dbo.lotes.cod_sproceso = dbo.subproceso.cod_sproceso
WHERE (NOT (dbo.lotes.limite IS NULL)) AND (dbo.lotes.tipo_proceso <> 0) AND dbo.lotes.nombre = '012312457' 
ORDER BY IdLote DESC;
