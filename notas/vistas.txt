v_data_usuario

SELECT        dbo.usuarios_pst.cod_usuario, dbo.usuarios_pst.usuario, dbo.usuarios_pst.nombre + ' ' + dbo.usuarios_pst.apellido AS nombre, dbo.usuarios_pst.pass, dbo.usuarios_pst.cod_rol, dbo.roles.nombre_rol AS rol
FROM            dbo.usuarios_pst INNER JOIN
                         dbo.roles ON dbo.roles.cod_rol = dbo.usuarios_pst.cod_rol

v_planilla_pst


SELECT        dbo.planillas_pst.cod_planilla, bdsystem.dbo.lotes.nombre AS lote, dbo.planillas_pst.fec_turno, bdsystem.dbo.turno.NomTurno AS turno, bdsystem.dbo.empresas.descripcion AS empresa, 
                         bdsystem.dbo.proveedores.descripcion AS proveedor, bdsystem.dbo.especies.descripcion AS especie, dbo.planillas_pst.cod_planillero, planillero.nombre + ' ' + planillero.apellido AS planillero_nombre, 
                         dbo.planillas_pst.cod_supervisor, supervisor.nombre + ' ' + supervisor.apellido AS supervisor_nombre, dbo.planillas_pst.guardado
FROM            dbo.planillas_pst LEFT OUTER JOIN
                         bdsystem.dbo.lotes ON dbo.planillas_pst.cod_lote = bdsystem.dbo.lotes.cod_lote LEFT OUTER JOIN
                         bdsystem.dbo.turno ON dbo.planillas_pst.cod_turno = bdsystem.dbo.turno.CodTurno LEFT OUTER JOIN
                         bdsystem.dbo.empresas ON dbo.planillas_pst.cod_empresa = bdsystem.dbo.empresas.cod_empresa LEFT OUTER JOIN
                         bdsystem.dbo.detalle_lote ON dbo.planillas_pst.cod_lote = bdsystem.dbo.detalle_lote.cod_lote LEFT OUTER JOIN
                         bdsystem.dbo.proveedores ON bdsystem.dbo.detalle_lote.cod_proveedor = bdsystem.dbo.proveedores.cod_proveedor LEFT OUTER JOIN
                         bdsystem.dbo.especies ON dbo.planillas_pst.cod_especie = bdsystem.dbo.especies.cod_especie LEFT OUTER JOIN
                         dbo.usuarios_pst AS planillero ON dbo.planillas_pst.cod_planillero = planillero.cod_usuario LEFT OUTER JOIN
                         dbo.usuarios_pst AS supervisor ON dbo.planillas_pst.cod_supervisor = supervisor.cod_usuario

v_registro_planilla_pst

CREATE VIEW v_registro_planilla_pst AS 

SELECT
	rp.cod_reg,
    rp.cod_planilla,
    ini.nombre AS cInicial,
    fin.nombre AS cFinal,
    s.nombre AS sala,
	de.nombre AS destino,
    c.nombre AS calibre,
    ca.nombre AS calidad,
    rp.piezas,
    rp.kilos,
    rp.guardado
FROM
    dbo.registro_planilla_pst AS rp
LEFT OUTER JOIN
    dbo.corte AS ini ON rp.cod_corte_ini = ini.cod_corte
LEFT OUTER JOIN
    dbo.corte AS fin ON rp.cod_corte_fin = fin.cod_corte
LEFT OUTER JOIN
    dbo.sala AS s ON rp.cod_sala = s.cod_sala
LEFT OUTER JOIN
    dbo.calibre AS c ON rp.cod_calibre = c.cod_calib
LEFT OUTER JOIN
    dbo.calidad AS ca ON rp.cod_calidad = ca.cod_cald
LEFT OUTER JOIN 
	dbo.destino AS de ON rp.cod_destino = de.cod_destino;
